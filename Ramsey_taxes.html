# Dynamic equations

def Resource_Constraint(k, c, _c):
    
    k_n = f(k) + (1-)*k - g - c
    
    return k_n

def Interest_Rate(k):
    return df_dk(k) - 

def Euler_Equation(k, c, _c, _c_n):
    
    k_n = Resource_Constraint(k, c, _c)
    
    if k_n > 0:
        r_n = Interest_Rate(k_n)
        c_n = ( (1+_c)/(1+_c_n) *  * (1+r_n) )**(1/) * c
    else:
        c_n = 0
        
    return c_n

# Forward equations and convergence criterion

def Path(c_0, k_0, T=20, T=50):
    
    T += 1
    
    k_t = np.zeros(T)
    c_t = np.zeros(T)
    _c_t = np.zeros(T)
    
    _c_t[:T] = _c_0
    _c_t[T:] = _c_1
    
    k_t[0] = k_0
    c_t[0] = c_0
    
    for t in range(T-1):
        k_t[t+1] = Resource_Constraint(k_t[t], c_t[t], _c_t[t])
        if k_t[t+1] > 0:
            c_t[t+1] = Euler_Equation(k_t[t], c_t[t], _c_t[t], _c_t[t+1])
        else:
            k_t[t+1] = 0
            c_t[t+1] = 0
            
    return k_t, c_t

def Path_crit(c_0, k_0, T=20, T=50):
    
    k_t, c_t = Path(c_0, k_0, T, T)
    
    ss_diff = np.sqrt((k_t-k_ss_1)**2 + (c_t-c_ss_1)**2)
    
    return np.min(ss_diff) + ss_diff[-1]
